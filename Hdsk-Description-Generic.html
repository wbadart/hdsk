genericMean :: (Real r, Fractional p) => [r] -> p Source #

Computes the arithmetic mean of a list of real numbers.

>>> mean [1, 3, 3, 4]
2.75
>>> mean [10]
10.0

genericVar :: (Real p, Floating p) => [p] -> p Source #

Computes the variance of a list of numbers.

>>> var [1, 2]
0.25
>>> var [10.12, 10.12]
0.0
